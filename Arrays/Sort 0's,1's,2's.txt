import java.util.*;

class Solution {
    public void sort012(int[] arr) 
    {
        ArrayList<Integer> list = new ArrayList<>();

        for(int num : arr)
        {
            list.add(num);
        }

        Collections.sort(list);

        for(int i = 0; i < arr.length; i++)
        {
            arr[i] = list.get(i);
        }
    }
}

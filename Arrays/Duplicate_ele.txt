class Solution {
    public int findDuplicate(int[] nums)
     {
        HashSet<Integer> seen=new HashSet<>();
        HashSet<Integer> dup=new HashSet<>();
        int x=0;
        for(int num:nums)
        {
            if(!seen.add(num))
            {
                 x=num;
            }
        }
        return x;
        
    }
}

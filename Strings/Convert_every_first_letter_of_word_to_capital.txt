public class Main
{
	public static void main(String[] args) 
	{
		String s="hii hello";
		char arr[]=s.toCharArray();
		int n=arr.length;
		arr[0]=Character.toUpperCase(arr[0]);
		for(int i=1;i<n;i++)
		{
		    if(arr[i-1]==' ')
		    {
		        arr[i]=Character.toUpperCase(arr[i]);
		    }
		}
		System.out.print(new String(arr));
		
	}
}

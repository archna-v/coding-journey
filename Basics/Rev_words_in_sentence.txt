class Solution {
    public String reverseWords(String s) 
    {
        String arr[]=s.split("\\.");
        ArrayList<String> list=new ArrayList<>();
        for(String str:arr)
        {
            if(!str.isEmpty())
            {
                list.add(str);
            }
        }
        Collections.reverse(list);
        return String.join(".",list);
        
    }
}

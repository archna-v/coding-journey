class Solution {
    static boolean armstrongNumber(int n)
    {
        int ac=n;
        int ac2=n;
        int rem;
        int res=0;
        int c=0;
        while(n!=0)
        {
            n=n/10;
            c++;
        }
        while(ac!=0)
        {
           rem=ac%10;
           res=res+(int)Math.pow(rem,c);
           ac=ac/10;
        }
        if(res==ac2) return true;
        return false;
        
    }
}

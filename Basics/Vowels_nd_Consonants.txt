public class Main
{
	public static void main(String[] args) 
	{
		String s="Aeghjou";
		char arr[]=s.toCharArray();
		int vow=0;
		int con=0;
		for(char ch:arr)
		{
		    if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u')
		    {
		        vow++;
		    }
		    else{
		        con++;
		    }
		}
		System.out.print(vow+" "+con);
	}
}
